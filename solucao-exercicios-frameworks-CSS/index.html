O flex-column aplicado ao <body> tem um efeito fundamental na estrutura e layout da página. Quando você aplica `d-flex flex-column` ao body, você está:

1. Tornando o body um container flexbox (d-flex)

2. Definindo a direção principal como vertical (flex-column)

Layout Resultante

<body class="d-flex flex-column min-vh-100">

<!-- Os elementos filhos se organizam verticalmente -->

<nav>...</nav> ← Topo

<main>...</main> ← Meio (cresce para preencher espaço)

<footer>...</footer> ← Rodapé fixo no final

</body>

Elementos-Chave que Trabalham Juntos

1.min-vh-100

- Garante que o body ocupe pelo menos 100% da altura da viewport

- Mesmo com pouco conteúdo, a página terá altura total da tela

2. flex-grow-1 no <main>

- O conteúdo principal cresce automaticamente para preencher o espaço disponível

- Empurra o footer para o final da página

3. `mt-auto` no `<footer>`

- Aplica margem automática no topo

- Em conjunto com flexbox, "cola" o footer na parte inferior

Resultado Visual

Sem flex-column:

```

[Navbar]

[Main Content]

[Footer] ← Pode ficar "flutuando" no meio da tela

[Espaço vazio]

```

Com flex-column:

```

[Navbar]

[Main Content - expande conforme necessário]

[Footer] ← Sempre no final da tela

```

Vantagens Práticas

- Footer sempre posicionado corretamente (não flutua no meio da tela)

- Layout responsivo que se adapta ao conteúdo

- Sem JavaScript necessário para posicionamento do footer

- Funciona com qualquer quantidade de conteúdo

      // Estado simples do carrinho
      const cart = [];

      function atualizarBadge() {
        document.getElementById('badgeQtd').textContent = cart.length;
      }

      function formatBRL(n) {
        return n.toLocaleString('pt-BR', { style: 'currency', currency: 'BRL' });
      }

      function atualizarCarrinhoUI() {
        const lista = document.getElementById('listaCarrinho');
        if (cart.length === 0) {
          lista.innerHTML = '<p class="text-body-secondary">Seu carrinho está vazio.</p>';
          document.getElementById('totalCarrinho').textContent = 'R$ 0,00';
          return;
        }
        let total = 0;
        lista.innerHTML = '<ul class="list-group mb-3" id="ulCarrinho"></ul>';
        const ul = document.getElementById('ulCarrinho');
        cart.forEach((item, idx) => {
          total += item.price;
          const li = document.createElement('li');
          li.className = 'list-group-item d-flex justify-content-between align-items-center';
          li.innerHTML = `<span>${item.name}</span><span class="fw-semibold">${formatBRL(item.price)}</span>`;
          ul.appendChild(li);
        });
        document.getElementById('totalCarrinho').textContent = formatBRL(total);
      }

      function showToast() {
        const toastEl = document.getElementById('toastOk');
        const toast = new bootstrap.Toast(toastEl, { delay: 1500 });
        toast.show();
      }

      // Adicionar ao carrinho
      document.querySelectorAll('.btn-add').forEach(btn => {
        btn.addEventListener('click', () => {
          const name = btn.getAttribute('data-product');
          const price = parseFloat(btn.getAttribute('data-price'));
          cart.push({ name, price });
          atualizarBadge();
          atualizarCarrinhoUI();
          showToast();
        });
      });

      // Preencher modal de detalhes dinamicamente
      const modalDetalhes = document.getElementById('modalDetalhes');
      modalDetalhes.addEventListener('show.bs.modal', event => {
        const button = event.relatedTarget;
        const name = button.getAttribute('data-product');
        const price = parseFloat(button.getAttribute('data-price'));
        document.getElementById('tituloModal').textContent = name;
        document.getElementById('descricaoModal').textContent = `O ${name} é excelente para demonstrar componentes do Bootstrap.`;
        document.getElementById('precoModal').textContent = price.toFixed(2).replace('.', ',');
      });

      // Offcanvas: atualizar UI ao abrir
      const carrinho = document.getElementById('carrinhoOffcanvas');
      carrinho.addEventListener('show.bs.offcanvas', atualizarCarrinhoUI);
    </script>
